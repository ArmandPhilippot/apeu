## Using callback functions to access IDs

<CodePreview
  code={`<Popover controllerId="callback-controller" modalId="callback-modal">
    <fragment slot="label">
    {
      (controllerId: string) => (
        <label for={controllerId}>Open callback example modal</label>
      )
    }
    </fragment>
    <fragment slot="modal">
    {
      (modalId: string) => (
        <div id={modalId} slot="modal">
          The content of the callback example modal.
        </div>
      )

    </fragment>
</Popover>`}
