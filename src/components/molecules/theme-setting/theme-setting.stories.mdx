import { components as mdxComponents } from "../../mdx";
import ThemeSetting from "./theme-setting.astro";

export const components = mdxComponents;
export const switchLabel = "Enable dark mode?";
export const toggleLabel = "Choose a theme:";

## Usage

A <code>ThemeSetting</code> component is used to let the user choose the he wants to use for the website.

<ThemeSetting label={toggleLabel} setting="theme" />

## Options

### <code>isInline</code>

<ThemeSetting
  id="inlined-example"
  isInline
  label={toggleLabel}
  setting="theme"
/>

### <code>variant="switch"</code>

<ThemeSetting  label={switchLabel} setting="theme" variant="switch" />

## With multiple toggles on the page

If you need multiple toggles on the same page for the same setting key, you must provide an <code>id</code>.

<ThemeSetting
  id="toggle1"
  label={toggleLabel}
  setting="shiki"
  variant="toggle"
/>

<ThemeSetting
  id="toggle2"
  label={toggleLabel}
  setting="shiki"
  variant="toggle"
/>

<ThemeSetting
  id="toggle3"
  label={toggleLabel}
  setting="shiki"
  variant="toggle"
/>

## With multiple switches on the page

If you need multiple switches on the same page for the <code>id</code> is not required.

<ThemeSetting label={switchLabel} setting="shiki" variant="switch" />

<ThemeSetting label={switchLabel} setting="shiki" variant="switch" />

<ThemeSetting label={switchLabel} setting="shiki" variant="switch" />
